\newcommand{\tikzCircleXLabel}[4]{
  % #1 = x, #2 = y, #3 = color, #4 = label
  \draw[thick, #3] (#1,#2) circle (0.2);
  \draw[thick, #3] (#1,#2) -- ({#1+0.141},{#2+0.141});
  \draw[thick, #3] (#1,#2) -- ({#1-0.141},{#2+0.141});
  \draw[thick, #3] (#1,#2) -- ({#1+0.141},{#2-0.141});

  \node[text=#3] at ({#1+0.9},{#2}) {#4};
}
